<script dangerouslySetInnerHTML={{
  __html: `
    console.log('Script started');
    
    function updateTheme() {
      console.log('updateTheme called');
      
      const isDark = document.documentElement.classList.contains('dark') || 
                     localStorage.getItem('mintlify-theme') === 'dark' ||
                     document.body.classList.contains('dark');
      
      console.log('Theme check:', { 
        isDark, 
        htmlClass: document.documentElement.className, 
        bodyClass: document.body.className, 
        storage: localStorage.getItem('mintlify-theme') 
      });
      
      const container = document.getElementById('overview-container');
      if (container) {
        console.log('Container found, updating styles');
        if (isDark) {
          container.style.background = '#0F172A';
          container.style.color = '#E2E8F0';
        } else {
          container.style.background = '#FFFFFF';
          container.style.color = '#1F2937';
        }
        
        // 更新卡片样式
        const cards = container.querySelectorAll('.card');
        console.log('Found cards:', cards.length);
        cards.forEach(card => {
          if (isDark) {
            card.style.background = 'rgba(255, 255, 255, 0.05)';
            card.style.border = '1px solid rgba(255, 255, 255, 0.1)';
            card.style.color = '#E2E8F0';
          } else {
            card.style.background = 'rgba(0, 0, 0, 0.02)';
            card.style.border = '1px solid rgba(0, 0, 0, 0.1)';
            card.style.color = '#1F2937';
          }
        });
        
        // 更新标题颜色
        const titles = container.querySelectorAll('.card-title, h1');
        titles.forEach(title => {
          if (isDark) {
            title.style.color = '#E2E8F0';
          } else {
            title.style.color = '#1F2937';
          }
        });
        
        // 更新描述文字颜色
        const descriptions = container.querySelectorAll('.card-description, p');
        descriptions.forEach(desc => {
          if (isDark) {
            desc.style.color = '#94A3B8';
          } else {
            desc.style.color = '#6B7280';
          }
        });
      } else {
        console.log('Container not found');
      }
    }
    
    // 初始设置
    setTimeout(updateTheme, 500);
    
    // 监听主题切换
    const observer = new MutationObserver(() => {
      console.log('DOM mutation detected');
      setTimeout(updateTheme, 100);
    });
    
    observer.observe(document.documentElement, { 
      attributes: true, 
      attributeFilter: ['class'] 
    });
    
    observer.observe(document.body, { 
      attributes: true, 
      attributeFilter: ['class'] 
    });
    
    // 监听点击事件
    document.addEventListener('click', (e) => {
      if (e.target.closest('button') && (
          e.target.closest('button').innerHTML.includes('sun') ||
          e.target.closest('button').innerHTML.includes('moon') ||
          e.target.closest('button').getAttribute('aria-label')?.includes('theme')
        )) {
        console.log('Theme button clicked');
        setTimeout(updateTheme, 200);
      }
    });
    
    // 定期检查
    setInterval(() => {
      updateTheme();
    }, 2000);
  `
}} />
