<script dangerouslySetInnerHTML={{
  __html: `
    function updateTheme() {
      const isDark = document.documentElement.classList.contains('dark') || 
                     localStorage.getItem('mintlify-theme') === 'dark' ||
                     document.body.classList.contains('dark');
      const container = document.getElementById('overview-container');
      if (!container) return;
      if (isDark) {
        container.style.background = '#0F172A';
        container.style.color = '#E2E8F0';
      } else {
        container.style.background = '#FFFFFF';
        container.style.color = '#1F2937';
      }
      const cards = container.querySelectorAll('.card');
      cards.forEach(card => {
        if (isDark) {
          card.style.background = 'rgba(255, 255, 255, 0.05)';
          card.style.border = '1px solid rgba(255, 255, 255, 0.1)';
          card.style.color = '#E2E8F0';
        } else {
          card.style.background = 'rgba(0, 0, 0, 0.02)';
          card.style.border = '1px solid rgba(0, 0, 0, 0.1)';
          card.style.color = '#1F2937';
        }
      });
      const titles = container.querySelectorAll('.card-title, h1');
      titles.forEach(title => {
        title.style.color = isDark ? '#E2E8F0' : '#1F2937';
      });
      const descriptions = container.querySelectorAll('.card-description, p');
      descriptions.forEach(desc => {
        desc.style.color = isDark ? '#94A3B8' : '#6B7280';
      });
    }
    if (document.readyState === 'loading') {
      document.addEventListener('DOMContentLoaded', updateTheme);
    } else {
      updateTheme();
    }
    const observer = new MutationObserver(updateTheme);
    observer.observe(document.documentElement, { attributes: true, attributeFilter: ['class'] });
    observer.observe(document.body, { attributes: true, attributeFilter: ['class'] });
    window.addEventListener('storage', (e) => { if (e.key === 'mintlify-theme') updateTheme(); });
  `
}} />
